cmake_minimum_required(VERSION 3.14)

project(neural_probe_tests LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(GTest REQUIRED)

add_executable(
    ring_buffer_tests
    tests/ring_buffer_test.cpp
)

target_link_libraries(ring_buffer_tests GTest GTest::gtest_main)

add_executable(
    protocol_tests
    tests/protocol_test.cpp
)

target_link_libraries(protocol_tests GTest GTest::gtest_main)

enable_testing()
add_test(NAME ring_buffer_tests COMMAND ring_buffer_tests)
add_test(NAME protocol_tests COMMAND protocol_tests)
